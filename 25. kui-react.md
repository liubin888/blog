#### 安装
```
npm install react-kui
```

#### 使用

```js
import React from 'react'
import Render from 'react-dom'
import { Button } from 'react-kui'
import 'react-kui/dist/k-ui.css'

Render.render(<Button type="primary"></Button>,document.getElementById('app'))
